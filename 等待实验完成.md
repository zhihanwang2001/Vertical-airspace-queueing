import json
import numpy as np
results = []
for seed in [42, 43, 44]:
    with open(f'Data/ablation_studies/hca2c_wide/hca2c_wide_seed{seed}_results.json') as f:
        results.append(json.load(f))
rewards = [r['mean_reward'] for r in results]
crashes = [r['crash_rate'] for r in results]
print(f'Mean: {np.mean(rewards):.1f} ± {np.std(rewards):.1f}')
print(f'Crash rate: {np.mean(crashes)*100:.0f}%')
print(f'CV: {np.std(rewards)/np.mean(rewards)*100:.2f}%' if np.mean(rewards) > 0 else 'CV: N/A')
"
